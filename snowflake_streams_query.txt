MERGE INTO SALES_FINAL_TABLE T
USING (
    SELECT
        S.ID,
        S.PRODUCT,
        S.PRICE,
        S.AMOUNT,
        S.STORE_ID,
        ST.LOCATION,
        ST.EMPLOYEES
    FROM SALES_STREAM S
    LEFT JOIN STORE_TABLE ST
        ON S.STORE_ID = ST.STORE_ID
    WHERE METADATA$ACTION = 'INSERT'
) SRC

ON T.ID = SRC.ID

WHEN MATCHED THEN
UPDATE SET
    T.PRODUCT = SRC.PRODUCT,
    T.PRICE = SRC.PRICE,
    T.AMOUNT = SRC.AMOUNT,
    T.STORE_ID = SRC.STORE_ID,
    T.LOCATION = SRC.LOCATION,
    T.EMPLOYEES = SRC.EMPLOYEES

WHEN NOT MATCHED THEN
INSERT (
    ID,
    PRODUCT,
    PRICE,
    AMOUNT,
    STORE_ID,
    LOCATION,
    EMPLOYEES
)
VALUES (
    SRC.ID,
    SRC.PRODUCT,
    SRC.PRICE,
    SRC.AMOUNT,
    SRC.STORE_ID,
    SRC.LOCATION,
    SRC.EMPLOYEES
);
