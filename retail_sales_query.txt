-- Use ACCOUNTADMIN role
USE ROLE ACCOUNTADMIN;

-- Create Database
CREATE DATABASE RETAIL_DB;

-- Create Schema
CREATE SCHEMA RETAIL_DB.ORDERS_DEMO;

-- Create Warehouse
CREATE WAREHOUSE RETAIL_LOAD_WH
WITH
WAREHOUSE_SIZE = 'XSMALL'
AUTO_SUSPEND = 60
AUTO_RESUME = TRUE;

-- Create Role
CREATE ROLE RETAIL_DATA_ENGINEER_ROLE;

-- Create User
CREATE USER RETAIL_ENGINEER_USER
PASSWORD = 'Retail@123'
DEFAULT_WAREHOUSE = RETAIL_LOAD_WH
DEFAULT_ROLE = RETAIL_DATA_ENGINEER_ROLE
MUST_CHANGE_PASSWORD = FALSE;

-- Grant Warehouse Access
GRANT USAGE ON WAREHOUSE RETAIL_LOAD_WH
TO ROLE RETAIL_DATA_ENGINEER_ROLE;

-- Grant Database Access
GRANT USAGE ON DATABASE RETAIL_DB
TO ROLE RETAIL_DATA_ENGINEER_ROLE;

-- Grant Schema Access
GRANT USAGE ON SCHEMA RETAIL_DB.ORDERS_DEMO
TO ROLE RETAIL_DATA_ENGINEER_ROLE;

-- Grant Create Table Permission
GRANT CREATE TABLE ON SCHEMA RETAIL_DB.ORDERS_DEMO
TO ROLE RETAIL_DATA_ENGINEER_ROLE;

-- Assign Role to User
GRANT ROLE RETAIL_DATA_ENGINEER_ROLE
TO USER RETAIL_ENGINEER_USER;

USE ROLE ACCOUNTADMIN;

USE DATABASE RETAIL_DB;
USE SCHEMA ORDERS_DEMO;

CREATE OR REPLACE FILE FORMAT RETAIL_CSV_FORMAT
TYPE = CSV
SKIP_HEADER = 1
FIELD_OPTIONALLY_ENCLOSED_BY = '"';

CREATE OR REPLACE STORAGE INTEGRATION RETAIL_AZURE_INT
TYPE = EXTERNAL_STAGE
STORAGE_PROVIDER = AZURE
ENABLED = TRUE
AZURE_TENANT_ID = '65087c47-0017-4258-8086-72832006d566'
STORAGE_ALLOWED_LOCATIONS = (
'azure://snowflakelearnretail.blob.core.windows.net/retail-data'
);

DESC INTEGRATION RETAIL_AZURE_INT;

CREATE OR REPLACE STAGE RETAIL_AZURE_STAGE
URL = 'azure://snowflakelearnretail.blob.core.windows.net/retail-data/raw/orders/'
CREDENTIALS = (
AZURE_SAS_TOKEN='?sv=2026-02-06&ss=bfqt&srt=co&sp=rwdlacupiytfx&se=2026-05-21T17:46:58Z&st=2026-05-21T09:31:58Z&spr=https&sig=9LMtaMsJcwFKcQk4ebXB00DoOc%2FLM7U9d2ZY14w92tI%3D'
)
FILE_FORMAT = RETAIL_CSV_FORMAT;

LIST @RETAIL_AZURE_STAGE;

CREATE OR REPLACE TABLE ORDERS_RAW (
    ORDER_ID NUMBER,
    CUSTOMER_NAME STRING,
    PRODUCT STRING,
    AMOUNT NUMBER,
    ORDER_DATE DATE
);

CREATE OR REPLACE TABLE ORDERS_FINAL (
    ORDER_ID NUMBER,
    CUSTOMER_NAME STRING,
    PRODUCT STRING,
    AMOUNT NUMBER,
    ORDER_DATE DATE
);

COPY INTO ORDERS_RAW
FROM @RETAIL_AZURE_STAGE
FILE_FORMAT = (FORMAT_NAME = RETAIL_CSV_FORMAT);

SELECT * FROM ORDERS_RAW;

INSERT INTO ORDERS_FINAL
SELECT * FROM ORDERS_RAW;

SELECT * FROM ORDERS_FINAL;

CREATE OR REPLACE PIPE RETAIL_ORDERS_PIPE
AUTO_INGEST = FALSE
AS
COPY INTO ORDERS_RAW
FROM @RETAIL_AZURE_STAGE
FILE_FORMAT = (FORMAT_NAME = RETAIL_CSV_FORMAT);

CREATE OR REPLACE STREAM ORDERS_RAW_STREAM
ON TABLE ORDERS_RAW;

CREATE OR REPLACE TASK ORDERS_MERGE_TASK
WAREHOUSE = RETAIL_LOAD_WH
SCHEDULE = '1 MINUTE'
AS
MERGE INTO ORDERS_FINAL T
USING (
    SELECT * FROM ORDERS_RAW_STREAM
) S
ON T.ORDER_ID = S.ORDER_ID

WHEN NOT MATCHED THEN
INSERT (
    ORDER_ID,
    CUSTOMER_NAME,
    PRODUCT,
    AMOUNT,
    ORDER_DATE
)
VALUES (
    S.ORDER_ID,
    S.CUSTOMER_NAME,
    S.PRODUCT,
    S.AMOUNT,
    S.ORDER_DATE
);

ALTER TASK ORDERS_MERGE_TASK RESUME;

ALTER PIPE RETAIL_ORDERS_PIPE REFRESH;

SELECT * FROM ORDERS_RAW;

SELECT * FROM ORDERS_FINAL
ORDER BY ORDER_ID;

ALTER PIPE RETAIL_ORDERS_PIPE REFRESH;

SELECT COUNT(*) FROM ORDERS_FINAL;
